#### docker-compose.yaml

```
version: '3.6'

networks:
  HW03_network:

services:
  HW03_MySQL:
    image: mysql:latest
    restart: unless-stopped
    container_name: HW03_MySQL
    command: 
      --default-authentication-plugin=mysql_native_password 
      --character-set-server=utf8 
      --collation-server=utf8_general_ci
    environment:
      TZ: Europe/Moscow
      MYSQL_ROOT_HOST: '%'
      MYSQL_DATABASE: 'homework03'
      MYSQL_USER: 'HW03_admin'
      MYSQL_PASSWORD: 'HW03_admin_password'
      MYSQL_ROOT_PASSWORD: 'db-root_password'
    ports:
      - '13306:3306'
    volumes:
      - ./HW03_MySQL/base:/var/lib/mysql
    networks:
      HW03_network:

  HW03_phpmyadmin:
    image: phpmyadmin:latest
    restart: unless-stopped
    container_name: HW03_phpMyAdmin
    ports:
      - 28080:80
    environment:
      - PMA_ARBITRARY=1
      - PMA_USER=root
      - PMA_PASSWORD=db-root_password
      - PMA_VERBOSES=HW03_MySQL
      - PMA_HOSTS=HW03_MySQL      
      - UPLOAD_LIMIT=1G
      - MEMORY_LIMIT=512m
    links:
      - HW03_MySQL
    depends_on:
      - HW03_MySQL
    networks:
      HW03_network:
```
